class Solution {
    public List<List<Integer>> minimumAbsDifference(int[] arr) {
        Arrays.sort(arr);
        int minVal = Integer.MAX_VALUE, n = arr.length;
        List<List<Integer>> result = new ArrayList<>();
        for (int i = 0; i < n - 1; i++) {
            int val = arr[i + 1] - arr[i];
            if (val < minVal) {
                minVal = val;
                result.clear();
                result.add(Arrays.asList(arr[i], arr[i + 1]));
            } else if (val == minVal) {
                result.add(Arrays.asList(arr[i], arr[i + 1]));
            }
        }
        return result;
    }
}
