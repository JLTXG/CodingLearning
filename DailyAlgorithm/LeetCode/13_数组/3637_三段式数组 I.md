class Solution {
    public boolean isTrionic(int[] nums) {
        int n = nums.length, i = 1;
        while (i < n && nums[i] > nums[i - 1]) {
            i++;
        }
        if (i == 1) {
            return false;
        }

        int iTwo = i;
        while (i < n && nums[i] < nums[i - 1]) {
            i++;
        }
        if (i == iTwo || i == n) {
            return false;
        }

        while (i < n && nums[i] > nums[i - 1]) {
            i++;
        }

        return i == n;
    }
}
