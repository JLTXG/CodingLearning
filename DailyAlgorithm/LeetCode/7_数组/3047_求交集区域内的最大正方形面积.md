class Solution:

        max_side = 0

                bx2, by2 = bottomLeft[j]
                tx2, ty2 = topRight[j]
                width = min(tx1, tx2) - max(bx1, bx2)
                height = min(ty1, ty2) - max(by1, by2)
                side = min(width, height)
                max_side = max(side, max_side)
        return max_side * max_side

class Solution {
    public long largestSquareArea(int[][] bottomLeft, int[][] topRight) {
        int maxSide = 0, n = bottomLeft.length;
        for (int i = 0; i < n; i++) {
            int[] bl1 = bottomLeft[i];

    }
