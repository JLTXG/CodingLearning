```Java
class Solution {
    public int[] minBitwiseArray(List<Integer> nums) {
        int n = nums.size();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            int tmp = nums.get(i);
            int res = -1, flag = 1;
            while ((tmp & flag) != 0) {
                res = tmp - flag;
                flag <<= 1;
            }
            arr[i] = res;
        }
        return arr;
    }
}
```

class Solution:
    def minBitwiseArray(self, nums: List[int]) -> List[int]:
        n = len(nums)
        arr = [0] * n
        for i in range(n):
            tmp, res, flag = nums[i], -1, 1
            while (tmp & flag) != 0 :
                res = tmp - flag
                flag <<= 1
            arr[i] = res
        return arr
