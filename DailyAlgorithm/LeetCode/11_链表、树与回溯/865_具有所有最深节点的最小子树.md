 * struct TreeNode {
 *     int val;
 *     TreeNode *left;

 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}

class Solution {
private:
    pair<TreeNode*, int> dfs(TreeNode* node) {
        if (node == nullptr) {
            return {nullptr, 0};
